#include<bits/stdc++.h>
using namespace std;

void FCFS(vector <int> a,int n, int h){
    int sum=0;
    for(int i=0;i<n;i++){
        cout<<a[i];
        if(i!=n-1){
            cout<<" -> ";
        }
        sum += abs(a[i]-h);
        h = a[i];
    }
    cout<<"\n\n\tTotal Seek Time : "<<sum;
}
