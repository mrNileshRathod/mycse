int strp=0,stkp=0;
char str[20],stack[20];
void reduce()
{
int i,stack_len;
char str1[]="E+E",str2[]="E*E";
stack_len=strlen(stack);
for(i=0;i< stack_len;i++)
{
if(stack[i]=='a'){
stack[i]='E';
}
if(stack[i]=='b'){
stack[i]='E';
}

}
if(strcmp(stack,str1)==0){
strcpy(stack,"E");

}
if(strcmp(stack,str2)==0){

}
}
void main()
{
