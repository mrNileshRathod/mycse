// Implement a C program for 3 address code for the input "a+b*c"
#include< stdio.h>
#include< string.h>
void pm();
void plus();
int i,len;
char expe[10],exp1[10],expe2[10];
void main()
{
printf("Enter the Expression:: ");
scanf("%s",expe);
strcpy(exp1,expe);
len=strlen(exp1);
expe2[0]='\0';
for(i=0;i< len;i++)
{
if((exp1[i]=='+') || exp1[i]=='-'){
if(exp1[i+2]=='*'){
pm();
break;
}
else{
plus();
break;
}
}
}
}
void pm()
{
strrev(exp1);
int j=len-i-1;
strncat(expe2,exp1,j);
strrev(expe2);
printf("\n\tThree Address Code :: \n\ttemp=%s \n\ttemp1=%c%ctemp\n",expe2,exp1[j+1],exp1[j]);
}
void plus()
{
strncat(expe2,exp1,i+2);
printf("\n\tThree Address Code :: \n\t\ttemp=%s \n\t\ttemp1=temp%c%c\n",expe2,exp1[i+2],exp1[i+3]);
