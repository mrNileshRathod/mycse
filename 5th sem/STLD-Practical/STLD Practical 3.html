library IEEE;
use IEEE.std_logic_1164.all;


entity FA is
port(
a,b,c: in bit;
sum,carry:out bit
);
end FA;


architecture arc_fa of FA is
begin
sum <= c xor a xor b;
carry <= (a and b) or (c and (a xor b));
end arc_fa;
