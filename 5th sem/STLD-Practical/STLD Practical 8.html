library IEEE;
use IEEE.std_logic_1164.all;
entity binary_to_gray is
port(
bin : in std_logic_vector(3 downto 0);
G : out std_logic_vector(3 downto 0)
);
end binary_to_gray;
architecture arc_binary_gray of binary_to_gray is
begin
G(3) <= bin(3);
G(2) <= bin(3) xor bin(2);
G(1) <= bin(2) xor bin(1);
G(0) <= bin(1) xor bin(0);
end arc_binary_gray;
