library IEEE;
use IEEE.std_logic_1164.all;
entity COMPARATOR_SOURCE is
port(
A : in STD_LOGIC_VECTOR (1 downto 0);
G,L,E : out STD_LOGIC
);
end COMPARATOR_SOURCE;
architecture arc_comparator of COMPARATOR_SOURCE is
begin
process(A)
begin
G<='0';
L<='0';
E<='0';
if(A(0) < A(1)) then
L<='1';
elsif(A(0) > A(1)) then
G<='1';
else
E<='1';
end if;
end process;
end arc_comparator;
