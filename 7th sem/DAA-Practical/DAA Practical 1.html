  #include<stdio.h> 

  void linearSearch(int a[],int n, int x) {	 
    int i,t; 
    for(i=0;i<n;i++) { 
      if(a[i]==x){ 
        t=i; 
        break; 
        } 
      } 
    if(t!=0) 
    printf("\nElement is found at location :%d",i+1); 
    else 
    printf("\nSorry !! Element is not found\n\n"); 
    } 

  void binarySerach(int a[],int n, int no){ 
    int f=0; 
    int l=n-1; 
    int mid = (f+l)/2; 
    
    while(f<=l) { 
      if(a[mid]<no) 
        f=mid+1; 
      else if(a[mid]==no) { 
        printf("%d is found at %d \n",no,mid+1);	 
        break; 
        } 
      else 
        l=mid-1; 
      
      mid=(f+l)/2;     
      } 
    if(f>l) 
      printf("Not found !"); 
    }
