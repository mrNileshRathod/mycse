    #include<stdio.h>
    #define n 8
    int arr[n][n];
    int visited[n];

    void DFS(int i){
        int j;
        printf("%2d",i);
        visited[i] = 1;
        for(j=0;j<n;j++){
            if(visited[j]==0 && arr[i][j]==1)
                DFS(j);
        }
    }
    void main(){
        int i,j;
            
        printf("Enter the adjecency matrix of the graph ::\n");
        for(i=0;i<n;i++){
            for(j=0;j<n;j++)
                scanf("%d",&arr[i][j]);	
                
            visited[i]=0;
        }
        printf("\n\nTraversing Graph Output :: \n");
        DFS(0);
    }
