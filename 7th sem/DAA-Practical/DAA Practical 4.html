    #include<stdio.h>
    int n,i,j,step=1,a,b,u,v;
    int visited[10]={0},min,mincost=0,matrix[10][10];
    void main()
    {
        printf("Enter the no of nodes :: ");
        scanf("%d",&n);
        printf("\nEnter the adjanecy matrix :: \n");
        for(i=1;i<=n;i++){
            for(j=1;j<=n;j++){
                scanf("%d",&matrix[i][j]);
                if(matrix[i][j] == 0)
                    matrix[i][j] = 999;
            }
        }
        
        visited[1] = 1;
        
        while(step<n){
            for(i=1,min=999;i<=n;i++){
                for(j=1;j<=n;j++){
                    if(matrix[i][j]<min){
                        if(visited[i]!=0){
                            min = matrix[i][j];
                            a=u=i;
                            b=v=j;
                        }
                    }
                }
            }
            if(visited[u]==0 || visited[v]==0){
                printf("\n Edge %d:(%d %d) cost:%d",step++,a,b,min);
                mincost+=min;
                visited[b]=1;
            }
            matrix[a][b]=matrix[b][a]=999;
        }
        printf("\nMinimum cost :: %d",mincost);
    
    }
