    #include<stdio.h>
    #define V 4
    #define INF 9999
    
    void display(int g[][V]){
        int i,j;
        printf("Output of Floyd Algorithm :: \n\n");
        for(i=0;i<V;i++){
            for(j=0;j<V;j++){
                if(g[i][j] == INF){			
                    printf("%6s","INF");
                }
                else{
                    printf("%6d", g[i][j]);
                }
            }
            printf("\n");
        }
    }
    
    void floydAlgo(int g[][V]){
        int dist[V][V],i,j,k;
        
        for(i=0;i<V;i++){
            for(j=0;j<V;j++){
                dist[i][j] = g[i][j];
            }
        }
        
        for(k=0;k<V;k++){
            for(i=0;i<V;i++){
                for(j=0;j<V;j++){
                    if(dist[i][j] > dist[i][k] + dist[k][j])
                        dist[i][j] = dist[i][k] + dist[k][j];
                }
            }	
        }
        display(dist);
    }
    
    void main() {
        int graph[V][V]={{0,5,INF,10},{INF,0,3,INF},{INF,0,1,INF},{INF,INF,INF,0}};
        floydAlgo(graph);
    }
