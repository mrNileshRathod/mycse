    #include<stdio.h>
    #define N 5

    void swap(int *x, int *y) {
        int t = *x;
        *x = *y;
        *y = t;	
      }

    void bubbleSort(int a[]) {
        int sort = 0;
        int i,j;
        for(i=0;i<N-1;i++){
            for(j=0;j<N-i-1;j++) {
                if(a[j]>a[j+1]) {
                    swap(&a[j], &a[j+1]);
                    sort = 1;
                  }
              }
            if(!sort) break;
          }
       }

    void print(int a[]) {
        int i;
        
        printf("\n\nSorted Array is :: ");
        for(i=0;i<N;i++)
            printf("%d ",a[i]);
      }

    void main() {
        int a[N], i;

        printf("Enter the 5 elements : ");
        for(i=0;i<N;i++)
            scanf("%d",&a[i]);

        // Applying Bubble Sort Algorithm
        bubbleSort(a);

        // Printing Sorted Array
        print(a);
       }
